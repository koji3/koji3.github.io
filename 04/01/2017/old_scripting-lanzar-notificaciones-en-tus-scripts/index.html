tempmax=80
tempactual=$(acpi -t | egrep '[0-9]+.[0-9]' -o | cut -d"." -f1)

texto="Alerta: Fusión en el núcleo. Temperatura: $tempactual."
if [ $tempactual -ge $tempmax ]; then
   notify-send "$texto"        #Notificación
   echo -e "\a"                #beep
fi
